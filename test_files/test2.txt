    Questo programma calcola il fattoriale di un numero.

def calcolaFattoriale(integer numero):integer {
    integer res << 0;
    if numero = 0 or numero = 1 then {
        res << 1;
    } else {
        res << numero * calcolaFattoriale(numero - 1);
    }
    return res;
}
