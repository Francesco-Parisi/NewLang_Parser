	scelta <-- "Inserisci Operazione:";

	while scelta < 1 or scelta > 5 loop
	{
		 scelta <-- "Operazione non valida!, Inserisci Operazione:";
	}

def effettuaOperazione(): float {
	float a, b, risultato;

    a <-- "Inserisci primo operando:";
    b <-- "Inserisci secondo operando:";

	if scelta = 1 then
	{
		risultato << a+b;
	} else {
		if scelta = 2 then
		{
			risultato << a-b;
		} else {
			if scelta = 3 then
			{
				risultato << a*b;
			} else {
				if scelta = 4 then
				{
					risultato << a/b;
				} else {
					if scelta = 5 then
					{
						risultato << a^b;
					}
				}
			}
		}
	}
	return risultato;
}

def continua(): boolean {
	integer a,b, scelta;
	boolean b;

	scelta <-- "inserisci 1 per effettuare un'altra operazione, 0 per terminare:";

	if scelta = 1 then
	{
		b << true;
	} else {
		if scelta = 0 then
		{
			b << false;
		}
	}
	return b;
}
